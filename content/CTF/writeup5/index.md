    def reverse_alpha(c):
        if 'a' <= c <= 'z':

            return chr(ord('Z') - (ord(c) - ord('A')))
        return c
    digit_map = {str(i): str(9 - i) for i in range(10)}
    symbol_map = {'_': '=', '=': '_', '-': '+', '+': '-'}
    result = []
    for c in string:
        if c.isalpha():
            result.append(reverse_alpha(c))
        elif c in digit_map:
            result.append(digit_map[c])
        elif c in symbol_map:
            result.append(symbol_map[c])
        else:
            result.append(c)
    return ''.join(result)

        elif c == '*':
            return '_'

            return '!'

            return '-'
        else:
            return c
